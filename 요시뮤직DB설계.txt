create table member(
   id varchar2(20) not null,
   pass varchar2(20) not null,
   name varchar2(10) not null,
   tel varchar2(20) not null,
   nickname varchar2(20) not null,
   regidate date default sysdate not null,
   primary key(id)
);
create table board(
   num number primary key,
   title varchar2(200) not null,
    content varchar2(2000) not null,
   id varchar(20) references member (id),
   postdate date default sysdate ,
   visitcount number not null,
   likes NUMBER(6) DEFAULT 0 NOT NULL
);
CREATE TABLE comments (
    num NUMBER PRIMARY KEY,
    cnum NUMBER NOT NULL,
    id VARCHAR2(20) NOT NULL,
    nickname VARCHAR2(20) NOT NULL,
    content VARCHAR2(2000) NOT NULL,
    postdate DATE DEFAULT SYSDATE NOT NULL,
    likes NUMBER(6) DEFAULT 0 NOT NULL
);
create table notice(
   num number primary key,
   title varchar2(200) not null,
   content varchar2(2000) not null,
   id varchar2(10) references member (id),
   postdate date default sysdate not null,
   visitcount number(6)
);
create table userboardlike(
   member_id varchar2(20) references member (id),
   board_num number references board (num)
);
create table userlike (
   member_id varchar2(20),
   comments_num number
);
create table song(
   snum  number primary key,
   stitle varchar2(30),
   sname varchar2(30),
   artist varchar2(30),
   sphoto varchar2(200),
   url varchar2(200)
);
create table banner(
   num  number primary key,
   url varchar2(200),
   fName varchar2(20)   
);
